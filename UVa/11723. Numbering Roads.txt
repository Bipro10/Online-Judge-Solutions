#include<iostream>
#include<cstdio>
#include<cmath>

using namespace std;
int main(){
    int r, n , ans=0, c=0;
    while(scanf("%d %d", &r, &n)==2){
        if(r==0 && n==0)
            break;
        if(r>n*27) cout<<"Case "<<++c<<": impossible"<<endl;
        //else if(n>=r) cout<<1
        else {
            ans=ceil((double)r/(double)n);
            cout<<"Case "<<++c<<": "<<ans-1<<endl;
        }
    }

return 0;
}
//complexity: O(1)
